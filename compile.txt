# DANS LE DOSSIER SRC

# POUR COMPILER (creer le fichier build avant)
mkdir ../build/
javac  -d ../build/ -target 1.6 ij/ImageJ.java ij/plugin/*.java nom/tam/fits/*.java

# copier fichiers de langues et icones
mkdir ../build/ij/i18n/
mkdir ../build/images/
cp ij/i18n/*.properties ../build/ij/i18n/
cp images/* ../build/images/
cp ij/*.properties ../build/ij/
cp MANIFEST.MF ../build

# POUR CREER UN JAR
cd ../build
jar cvfm SalsaJ2.jar MANIFEST.MF ij/ images/ nom/

# POUR EXECUTER (dans build)
# creer dossier plugins et y copier le jar des plugins
mkdir plugins
cp path/to/SalsaJ_plugins.jar plugins/
java -jar SalsaJ2.jar 


# POUR EFFACER LES .class créés au cas ou
find . -name "*.class" -delete
